Circuit

Image 3. The final circuit, with three individually functional LEDs.

Code 

 
 void setup() {
  // initialize pin 13 as an output.
  pinMode(13, OUTPUT);
  // initialize pin 12 as an output.
  pinMode(12, OUTPUT);
  // initialize pin 11 as an output.
  pinMode(11, OUTPUT);

}

void loop() {
  // turn the Green LED on
  digitalWrite(13, HIGH);
  // wait .6 seconds
  delay(600);
  // turn the Yellow LED on
  digitalWrite(12, HIGH);
  // wait .6 seconds
  delay(600);
  // turn the Blue LED on
  digitalWrite(11, HIGH);
  // wait .6 seconds
  delay(600);
  // turn the Green LED off
  digitalWrite(13, LOW);
  // wait .2 seconds
  delay(200);
  // turn the Yellow LED off
  digitalWrite(12, LOW);
  // wait .2 seconds
  delay(200);
  // turn the Blue LED off
  digitalWrite(11, LOW);
  // wait .6 seconds
  delay(600);
  // turn the Blue LED on
  digitalWrite(11, HIGH);
  // wait .6 seconds
  delay(600);
  // turn the Blue LED off
  digitalWrite(11, LOW);
  // turn the Yellow LED on
  digitalWrite(12, HIGH);
  // wait .6 seconds
  delay(600);
  // turn the Yellow LED off
  digitalWrite(12, LOW);
  // turn the Green LED on
  digitalWrite(13, HIGH);
  // wait .6 seconds
  delay(600);
  // turn the Green LED off
  digitalWrite(13, LOW);
  // wait .6 seconds
  delay(600);
}
